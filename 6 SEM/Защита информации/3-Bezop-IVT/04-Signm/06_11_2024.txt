> with(numtheory):
> #
> # =======================
> # Алиса и Боб генерируют:
> # ----------------------------------------------------
> p:= nextprime(11200230000005430067343456735730000000);
> g:= primroot(p);
> #
> # -----------------------------------------------------------
> # Алиса вычисляет x, y, генерирует случайное r и вычисляет t:
> # -----------------------------------------------------------
> x:= nextprime(12312232323) mod p;
> y:= (g &^ x) mod p;
> r:= 12342323236789 mod p;
> t:= g &^ r mod p;
> # --------------------------------
> # Боб генерирует и посылает вызов:
> # -------------------------------------------------
> s:= 234452323891011 mod p;       # s - случайное...
> # -------------------------------------------------
> # Алиса вычисляет и отправляет Бобу: 
> # ----------------------------------
> u:= r + s*x;
> # --------------
> # Боб проверяет:
> # --------------------------
> (g &^ u - t * y &^ s) mod p;
> #
